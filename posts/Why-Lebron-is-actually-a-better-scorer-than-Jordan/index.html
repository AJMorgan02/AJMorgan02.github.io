Load necessary libraries

library(tidyverse) library(ggplot2) library(gt) library(scales) library(RColorBrewer)

Create data for Michael Jordan

jordan_data <- data.frame( season = c(“1985”, “1986”, “1987”, “1988”, “1989”, “1990”, “1991”, “1992”, “1993”, “1994-95”, “1995-96”, “1996-97”, “1997-98”, “2001-02”, “2002-03”), GP = c(82, 18, 82, 82, 81, 82, 82, 80, 78, 17, 82, 82, 82, 60, 82), PTS = c(28.2, 22.7, 37.1, 35.0, 32.5, 33.6, 31.5, 30.1, 32.6, 26.9, 30.4, 29.6, 28.7, 22.9, 20.0), FG_Percentage = c(51.5, 45.7, 48.2, 53.5, 53.8, 52.6, 53.9, 51.9, 49.5, 41.1, 49.5, 48.6, 46.5, 41.6, 44.5) )

Calculate cumulative points for each season

jordan_data <- jordan_data %>% mutate(cumulative_points = cumsum(PTS))

lebron_data <- lebron_data %>% mutate(cumulative_points = cumsum(PTS))

Data for milestone chart (example values)

milestones <- data.frame( Player = c(“Jordan”, “Jordan”, “Jordan”, “LeBron”, “LeBron”, “LeBron”), Milestone = c(10000, 20000, 30000, 10000, 20000, 30000), Games_Played = c(303, 620, 960, 368, 726, 1072) )

Plot cumulative points progression for Jordan and LeBron

ggplot() + geom_line(data = jordan_data, aes(x = season, y = cumulative_points, color = “Jordan”), group = 1, size = 1.2) + geom_line(data = lebron_data, aes(x = season, y = cumulative_points, color = “LeBron”), group = 1, size = 1.2) + labs( title = “Cumulative Points Progression: Jordan vs. LeBron”, x = “Season”, y = “Cumulative Points”, caption = “Source: AJ Morgan” ) + scale_color_manual(values = c(“Jordan” = “red”, “LeBron” = “blue”)) + theme_minimal()

Milestone progression chart

ggplot(milestones, aes(x = Games_Played, y = Milestone, color = Player)) + geom_step(size = 1.2) + labs( title = “Scoring Milestones: Jordan vs. LeBron”, x = “Games Played”, y = “Points”, caption = “Source: AJ Morgan” ) + scale_color_manual(values = c(“Jordan” = “red”, “LeBron” = “blue”)) + theme_minimal()

Plot Field Goal Percentage over Seasons for Jordan and LeBron
